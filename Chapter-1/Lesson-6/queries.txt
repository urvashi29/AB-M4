create database transactions

use transactions

create table employee (
empId int primary key,
name varchar(40),
age int,
city varchar(50),
income int
)

insert into employee values(1, "alina", 23, "pune", 50000)

select * from employee;


create table orders (
orderId int primary key,
productName varchar(40),
orderNumber int
)

start transaction;

insert into orders values(1, "Laptop", 901)
select * from orders;

commit;

-- rollback;
start transaction;
delete from employee where empId = 1;
rollback;

select * from employee

-- savepoint 
start transaction;
insert into orders values (2, "Printer", 200)
insert into orders values (3, "Laptop", 500)

savepoint mysavepoint_1;

insert into orders values (4, "TV", 550)
savepoint mysavepoint_2;
select * from orders;

rollback to savepoint mysavepoint_1;
commit;

-- delete savepoint, but can't delete if transaction is commited.
release savepoint mysavepoint_1;
